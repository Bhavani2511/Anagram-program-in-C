char* rem_alph(char* s)
{
        int i,k=0;
        for(i=0;s[i];i++)
        {
                if((s[i]>='A' && s[i]<='Z') || (s[i]>='a' && s[i]<='z'))
                s[k++]=s[i];
        }
        s[k]='\0';
        return s;
}

char* convert_case(char* s)
{
        int i;
        for(i=0;s[i];i++)
        {
                if (s[i]>='A' && s[i]<='Z')
                        s[i]=s[i]+32;  // Convert to lowercase
        }
        return s;
}

char* bub_sort(char* s)
{
        int i,j,n,temp;
        n=strlen(s);
        for(i=0;i<n-1;i++)
        {
                for(j=0;j<n-i-1;j++)
                {
                        if(s[j]>s[j+1])
                        {
                                temp=s[j];
                                s[j]=s[j+1];
                                s[j+1]=temp;
                        }
                }
        }
        return s;
}
